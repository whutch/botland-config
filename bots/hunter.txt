// Suggested hardware: missiles 3, cloak 2, shield 2


update = function () {
    if (!init) {
        direction = "left";
        last_life = life;
        init = true;
    }
    if (life < last_life) {
        last_life = life;
        defend();
    }
    found = findClosestEnemyBot();
    if (exists(found)) {
        if (willMissilesHit(found)) {
            fireMissiles(found);
        }
        if (willArtilleryHit()) {
            fireArtillery();
        }
        pursue(found);
    }
    if (!canMove(direction)) {
        direction = next_direction(direction)
    }
    move(direction);
};

defend = function() {
    if (isShielded() || isCloaked() || isReflecting()) return;
    if (canShield()) shield();
    if (canCloak()) cloak();
    if (canReflect()) reflect();
}

next_direction = function(direction) {
    if (direction == "left") return "up";
    if (direction == "up") return "right";
    if (direction == "right") return "down";
    if (direction == "down") return "left";
};
